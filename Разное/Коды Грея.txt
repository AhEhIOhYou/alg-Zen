//Пока вводим переводим сразу в код Грея
// ^ - операция поразрядного ИЛИ, пример - было 101 и 011 стало -Ю 110 (1 - 0 = 1; 0 - 1 = 1; 1 - 1 = 0; 0 - 0 = 0)
// >> - Оператор побитового сдвига вправо сдвигает биты вправо, пример - 12 = 1100; 12 >> 1 = 0110 = 6
//Весь алгоритм сводится к одной формуле

int main() {
    unsigned long int k;
    while (cin >> k) {
        cout <<  (k  ^ (k >> 1)) << endl;
    }
    return 0;
}
