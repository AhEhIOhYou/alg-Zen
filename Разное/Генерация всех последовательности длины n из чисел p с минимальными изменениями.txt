#include <iostream>

//длина
#define n 4
//алфавит
#define k 3
//буферные массивы
int x[n],d[n];

//Вернет false если продолжать невозможно
int isContin(int i) {
//делааем ход и проверяем
    if ((x[i]==k-1 && d[i]==1) || (x[i]==0 && d[i]==-1))
        return 1;
    else
        return 0;
}

int main() {
    int i,j;
    for (i = 0; i < n; i++) {
        //Сначала все шашки на нижней горизонтали.
        x[i]=0;
        //Сначала все стрелки вверх.
        d[i]=1;
    }

    while(true) {
        //Печатаем очередную последовательность.
        for (i = 0; i < n; i++)
            cout << x[i];
        cout << endl;
        //Ищем самую правую шашку, которая может сделать ход.
        for (i = n - 1; i >= 0 && isContin(i); i--)
            if( i == -1 )
                //Если не нашли, то генерация закончена..
                break;

        x[i]=x[i]+d[i]; //Если нашли, делаем ход.
        //У шашек правее нашей меняем направление стрелок.
        for(j = i+1; j < n; j++)
            d[j]=-d[j];
    }
    return 0;
}
